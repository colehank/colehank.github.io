  // Make the "DOI" button reveal the DOI string instead of opening doi.org.
  (function () {
    document.querySelectorAll('.publications a.btn[href*="doi.org"]').forEach(function (a) {
      if (a.textContent.trim() !== "DOI") return; // leave HTML/PDF/other buttons alone
      var doi = a.getAttribute("href").replace(/^https?:\/\/(dx\.)?doi\.org\//i, "");
      a.setAttribute("title", "Show / hide the DOI");
      a.addEventListener("click", function (e) {
        e.preventDefault();
        var open = a.dataset.open === "1";
        a.textContent = open ? "DOI" : doi;
        a.dataset.open = open ? "0" : "1";
      });
    });
  })();
</script>
